public void appendStrToFile(String FileName,String str){		
    try {
	BufferedWriter out = new BufferedWriter(new FileWriter(FileName, true));
	out.write(str);
        out.newLine();
	out.close();
        }
	catch (IOException e) {
        System.out.println("exception occurred" + e);
	}
}

    private void ImportActionPerformed(java.awt.event.ActionEvent evt) {                                       
    try {
    FileReader  files = new FileReader(FileName);
    BufferedReader br = new BufferedReader(files);
    DefaultTableModel model = (DefaultTableModel)jTable.getModel();
    String line ; 
    String names[];
    ArrayList <String> row = new ArrayList<>();
    while ((line = br.readLine()) != null) { 
        names = line.split("\\p{javaWhitespace}+");
          if (names.length >= 2) {
                    String Name1 = names[0];
                    String Name2 = names[1];
                    String text = Name1 + " " + Name2;
                    row.add(text);
                }
            }
            br.close();
            SortingAlgo.bubbleSort(row, 0, row.size() -1);
             for (String text : row) {
                names = text.split("\\p{javaWhitespace}+");
                String Name1 = names[0];
                String Name2 = names[1];
                model.addRow(new Object[]{Name1,Name2});
             }
            
     }catch(IOException e){
     }                
    }                                      


 public class SortingAlgo {
    public static void bubbleSort(ArrayList <String> row , int i, int j) {
        for (i = 0; i < row.size()-1; i++){
            for ( j = i +1 ; j < row.size(); j++){
                if (row.get(j).compareToIgnoreCase(row.get(i))<0){
                    String temp = row.get(i);
                    row.set(i, row.get(j));
                    row.set(j, temp);
                }       
            }
        }
    }
 }
